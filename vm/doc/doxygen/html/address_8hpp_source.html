00023 /// An address of a constant series of bytes in memory.
00024 /**
00025  * One of the types that can be stored in Data.
00026  * 
00027  * \note This is just a wrapper for an Int8 constant pointer, with an explicit constructor.
00028  *       This is useful, because \c 0 cannot be interpreted as an Address (unlike <tt>Int8 *</tt>),
00029  *       you have to explicitly say <tt>Address(0)</tt>.
00030  *       This makes something like \code Data d = 0; \endcode unambiguous.
00031  *       (It's interpreted as a Number, not an Address.)
00032  */
00033 class Address {
00034     protected:
00035         Int8 const * address;
00036     public:
00037         inline explicit Address(Int8 const * const & address = 0) : address(address) {}
00038         inline operator Int8 const *       & ()       { return address; }
00039         inline operator Int8 const * const & () const { return address; }
00040 };
