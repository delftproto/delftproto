00080     void TUP_MAP(Machine & machine){
00081         Tuple & values = machine.stack.peek(0).asTuple();
00082         if (values.empty()){
00083             machine.stack.popTuple();
00084             machine.stack.popAddress();
00085             machine.stack.push(Tuple());
00086         } else {
00087             Address filter = machine.stack.peek(1).asAddress();

00122     /// Fold a Tuple to a single element with a given 'fuse function'.
00123     /**
00124      * The fuse function is used to consecutively fuse the previous fuse result with the next element of the Tuple.
00125      * The first element will be fused with the given starting value.
00126      * 
00127      * For example, a tuple with three elements is folded like this:
00128      * \dot
00129      * digraph Folding {
00130      *   node [fontsize=10,width=0,height=0];
00131      *   start -> a -> b -> c -> result;
00132      *   "first element" -> a;
00133      *   "second element" -> b;
00134      *   "third element" -> c;
00135      *   a [shape=box,label=fuse];
00136      *   b [shape=box,label=fuse];
00137      *   c [shape=box,label=fuse];
00138      * }
00139      * \enddot
00140      * 
00141      * If an empty Tuple is given, the starting value is returned.
00142      * 
00143      * \param Address The (address of the) fuse function. (The fold function must take two parameters.)
00144      * \param Data The value to start with.
00145      * \param Tuple The tuple to fold.
00146      * 
00147      * \return Data The result of the folding process.
00148      */
00149     void FOLD(Machine & machine){
00150         Tuple   fold_values = machine.stack.pop().asTuple();
00151         Data    result      = machine.stack.pop();
00152         Address fold_fuse   = machine.stack.peek().asAddress();
00153         
00154         if (fold_values.empty()){
00155             machine.stack.popAddress();
00156             machine.stack.push(result);
00157         } else {
00158             machine.stack.push(fold_values);
00159             machine.stack.push(0);
00160             machine.environment.push(result);
00161             machine.environment.push(fold_values[0]);
00162             machine.call(fold_fuse, fold_step);
00163         }
00164     }
